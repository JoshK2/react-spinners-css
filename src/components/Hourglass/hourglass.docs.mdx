import { Hourglass } from './hourglass';

Default color:

```js live
<Hourglass />
```

With custom color:

```js live
<>
  <Hourglass color="#be97e8" />
  <Hourglass color="#54f1d2" />
</>
```

Different sizes:

```js live
<>
  <Hourglass />
  <Hourglass size={32} />
  <Hourglass size={16} />
</>
```
